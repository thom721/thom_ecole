### Pagination — mise à jour des services

Ajoutez `skip` et `limit` dans `event_service.py` et `news_service.py` :

```python
# Dans get_all() des deux services
def get_all(
    db: Session,
    audience: Optional[AudienceType] = None,
    published_only: bool = False,
    skip: int = 0,
    limit: int = 10
):
    query = db.query(Event)  # ou News
    if audience:
        query = query.filter(Event.audience == audience)
    if published_only:
        query = query.filter(Event.is_published == True)
    
    total = query.count()
    items = query.order_by(Event.start_date).offset(skip).limit(limit).all()
    
    return {"total": total, "items": items}
```

Et dans les routes :

```python
@router.get("/", )
def list_events(
    audience: Optional[AudienceType] = None,
    published_only: bool = False,
    page: int = 1,
    limit: int = 10,
    db: Session = Depends(get_db)
):
    skip = (page - 1) * limit
    result = event_service.get_all(db, audience, published_only, skip, limit)
    return {
        "total": result["total"],
        "page": page,
        "limit": limit,
        "pages": (result["total"] + limit - 1) // limit,
        "data": result["items"]
    }
```
