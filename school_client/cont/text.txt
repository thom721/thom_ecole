```python
from PySide6.QtWidgets import QWidget, QVBoxLayout
from PySide6.QtWebEngineWidgets import QWebEngineView

class PDFViewer(QWidget):
    def __init__(self, pdf_base64):
        super().__init__()
        self.setWindowTitle("Aperçu du Bulletin")
        self.resize(800, 600)

        # Créer le navigateur intégré
        self.web_view = QWebEngineView()
        
        # Charger le PDF en tant qu'URL data
        pdf_html = f'<embed src="data:application/pdf;base64,{pdf_base64}" type="application/pdf" width="100%" height="100%"/>'
        self.web_view.setHtml(pdf_html)

        layout = QVBoxLayout()
        layout.addWidget(self.web_view)
        self.setLayout(layout)
```

### 2. **Modifie `_handle_response` pour détecter les PDF**

```python
def _handle_response(self, reply):
    endpoint = reply.property("endpoint")
    method = reply.property("method")
    try:
        raw_bytes = bytes(reply.readAll())
        status_code = reply.attribute(QNetworkRequest.HttpStatusCodeAttribute)

        content_type = reply.rawHeader(b"Content-Type").data().decode('utf-8', errors='ignore').lower()

        if 'application/pdf' in content_type:
            # Réponse binaire (PDF)
            self.request_binary_complete.emit(endpoint, method, raw_bytes)
            return

        try:
            raw_data = raw_bytes.decode()
            response_data = json.loads(raw_data) if raw_data else {}
        except UnicodeDecodeError:
            self.request_failed.emit(endpoint, method, "Erreur de décodage du contenu", {})
            return
        except json.JSONDecodeError as e:
            self.request_failed.emit(endpoint, method, f"Invalid JSON: {str(e)}", {})
            return

        if reply.error() and status_code != 200:
            error_msg = reply.errorString()
            self.request_failed.emit(endpoint, method, error_msg, response_data)
        else:
            self.request_complete.emit(endpoint, method, response_data)

    except Exception as e:
        self.request_failed.emit(endpoint, method, f"Unexpected error: {str(e)}", {})
    finally:
        reply.deleteLater()
```

def _handle_response(self, reply):
    endpoint = reply.property("endpoint")
    method = reply.property("method")
    
    try:
        raw_bytes = bytes(reply.readAll())
        status_code = reply.attribute(QNetworkRequest.HttpStatusCodeAttribute)
        
        # Récupération du Content-Type de manière compatible PySide6
        content_type = ""
        content_header = reply.rawHeader(b"Content-Type")
        if content_header:
            content_type = bytes(content_header).decode('utf-8', errors='ignore').lower()
        
        # Gestion des erreurs HTTP
        if reply.error():
            error_msg = reply.errorString()
            try:
                error_data = json.loads(raw_bytes.decode()) if raw_bytes else {}
            except:
                error_data = {}
            self.request_failed.emit(endpoint, method, error_msg, error_data, status_code)
            return

        # Gestion des différents types de contenu
        if 'application/pdf' in content_type:
            # Réponse binaire (PDF)
            self.request_binary_complete.emit(endpoint, method, raw_bytes, status_code)
        elif 'application/json' in content_type or not content_type:
            # Réponse JSON ou type inconnu (tentative de parsing JSON)
            try:
                response_data = json.loads(raw_bytes.decode()) if raw_bytes else {}
                self.request_complete.emit(endpoint, method, response_data, status_code)
            except json.JSONDecodeError:
                # Si le parsing JSON échoue mais que la requête a réussi
                if status_code == 200:
                    self.request_binary_complete.emit(endpoint, method, raw_bytes, status_code)
                else:
                    self.request_failed.emit(
                        endpoint, 
                        method, 
                        "Invalid JSON response", 
                        {"raw_response": raw_bytes.decode(errors='ignore')}, 
                        status_code
                    )
        else:
            # Autres types de contenu (texte, html, etc.)
            self.request_binary_complete.emit(endpoint, method, raw_bytes, status_code)
            
    except Exception as e:
        self.request_failed.emit(
            endpoint, 
            method, 
            f"Response handling error: {str(e)}", 
            {"exception": str(e)}, 
            status_code if 'status_code' in locals() else 0
        )
    finally:
        reply.deleteLater()
