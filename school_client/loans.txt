public function createLoan(Request $request)
{
    $P = $request->amount;
    $rate = $request->interest_rate / 100;
    $months = $request->duration_months;

    // Calcul prêts simples (sans amortissement complexe)
    $totalInterest = $P * $rate;
    $totalToPay = $P + $totalInterest;

    $monthlyPayment = $totalToPay / $months;

    $loan = Loan::create([
        'staff_id' => $request->staff_id,
        'amount' => $P,
        'interest_rate' => $request->interest_rate,
        'duration_months' => $months,
        'monthly_payment' => $monthlyPayment,
        'balance' => $totalToPay,
        'start_date' => now(),
    ]);

    // Génération des échéances mensuelles
    for ($i = 1; $i <= $months; $i++) {
        Installment::create([
            'loan_id' => $loan->id,
            'amount' => $monthlyPayment,
            'due_date' => now()->addMonths($i),
        ]);
    }

    return response()->json($loan);
}

✅ 4. Ajouter un paiement dans Laravel

public function payInstallment(Request $request)
{
    $installment = Installment::find($request->installment_id);
    $loan = $installment->loan;

    $payment = Payment::create([
        'loan_id'        => $loan->id,
        'installment_id' => $installment->id,
        'amount'         => $installment->amount,
        'payment_date'   => now()
    ]);

    // Marquer l'échéance comme payée
    $installment->paid = true;
    $installment->paid_date = now();
    $installment->save();

    // Réduire le solde du prêt
    $loan->balance -= $installment->amount;

    if ($loan->balance <= 0) {
        $loan->balance = 0;
        $loan->status = 'finished';
    }

    $loan->save();

    return response()->json([
        'loan' => $loan,
        'payment' => $payment
    ]);
}

4) Logique d’amortissement (PHP) — calcul mensuel

Simple amortissement (prêt amorti standard) — fonction helper :

app/Services/LoanService.php

namespace App\Services;

class LoanService
{
    // Calcule mensualité (annuité) donnée amount, annual rate (%) et term en mois
    public static function monthlyPayment(float $amount, float $annualRatePercent, int $termMonths): float
    {
        if ($annualRatePercent <= 0) {
            return round($amount / $termMonths, 2);
        }
        $monthlyRate = $annualRatePercent / 100 / 12;
        $factor = pow(1 + $monthlyRate, $termMonths);
        $payment = $amount * ($monthlyRate * $factor) / ($factor - 1);
        return round($payment, 2);
    }

    // Générer échéancier simple (tableau d'items: period, principal, interest, balance)
    public static function amortizationSchedule(float $amount, float $annualRatePercent, int $termMonths): array
    {
        $schedule = [];
        $balance = $amount;
        $monthlyPayment = self::monthlyPayment($amount, $annualRatePercent, $termMonths);
        $monthlyRate = $annualRatePercent / 100 / 12;

        for ($i=1; $i<= $termMonths; $i++) {
            $interest = round($balance * $monthlyRate, 2);
            $principal = round(min($monthlyPayment - $interest, $balance), 2);
            $balance = round($balance - $principal, 2);
            $schedule[] = [
                'period' => $i,
                'payment' => $monthlyPayment,
                'principal' => $principal,
                'interest' => $interest,
                'balance' => max($balance, 0),
            ];
        }

        return $schedule;
    }
}

Utilise monthlyPayment() lors de la création/approbation pour renseigner monthly_payment et remaining_balance.

namespace App\Http\Controllers\Api;

use App\Http\Controllers\Controller;
use Illuminate\Http\Request;
use App\Models\Loan;
use App\Models\LoanRepayment;
use App\Services\LoanService;
use Illuminate\Support\Facades\Auth;
use Illuminate\Validation\ValidationException;

class LoanController extends Controller
{
    // Lister prêts (pour admin ou user propre)
    public function index(Request $req)
    {
        $user = $req->user();
        if ($user->hasRole('admin')) {
            $loans = Loan::with('user','repayments')->latest()->paginate(20);
        } else {
            $loans = Loan::with('repayments')->where('user_id', $user->id)->latest()->get();
        }
        return response()->json($loans);
    }

    // Créer demande de prêt
    public function store(Request $req)
    {
        $req->validate([
            'amount' => 'required|numeric|min:1',
            'term_months' => 'required|integer|min:1',
            'interest_rate' => 'numeric|min:0',
            'notes' => 'nullable|string',
        ]);

        $loan = Loan::create([
            'user_id' => $req->user()->id,
            'amount' => $req->amount,
            'term_months' => $req->term_months,
            'interest_rate' => $req->interest_rate ?? 0,
            'status' => 'pending',
            'notes' => $req->notes,
        ]);

        return response()->json($loan, 201);
    }

    // Admin approuve et calcule mensualité
    public function approve(Request $req, Loan $loan)
    {
        $this->authorize('approve', $loan); // policy
        if ($loan->status !== 'pending') {
            return response()->json(['message'=>'Loan not pending'], 400);
        }
        $monthly = LoanService::monthlyPayment($loan->amount, $loan->interest_rate, $loan->term_months);
        $loan->update([
            'monthly_payment' => $monthly,
            'remaining_balance' => $loan->amount,
            'status' => 'approved',
            'approved_by' => $req->user()->id,
            'approved_at' => now(),
        ]);
        return response()->json($loan);
    }

    // Décaisser (disburse)
    public function disburse(Request $req, Loan $loan)
    {
        $this->authorize('disburse', $loan);
        if ($loan->status !== 'approved') {
            return response()->json(['message'=>'Loan not approved'], 400);
        }
        $loan->update([
            'status' => 'disbursed',
            'disbursed_at' => now(),
        ]);
        // ici tu pourrais générer l'échéancier et créer des lignes de "expected_payments" si tu veux persist.
        return response()->json($loan);
    }

    // Enregistrer un paiement
    public function repay(Request $req, Loan $loan)
    {
        $req->validate([
            'paid_amount' => 'required|numeric|min:0.01',
            'payment_date' => 'required|date',
            'payment_method' => 'nullable|string',
        ]);

        if (!in_array($loan->status, ['disbursed','default'])) {
            return response()->json(['message'=>'Loan is not disbursed'], 400);
        }

        $repayment = LoanRepayment::create([
            'loan_id' => $loan->id,
            'paid_amount' => $req->paid_amount,
            'payment_date' => $req->payment_date,
            'payment_method' => $req->payment_method,
            'collected_by' => $req->user()->id,
        ]);

        // mettre à jour remaining_balance
        $loan->remaining_balance = round($loan->remaining_balance - $req->paid_amount, 2);
        if ($loan->remaining_balance <= 0) {
            $loan->status = 'paid';
            $loan->remaining_balance = 0;
        }
        $loan->save();

        return response()->json($repayment, 201);
    }
}

8) Exemple client PySide6 (simplifié)

Utilise requests pour appeler API. Dans une vraie app, fais les appels dans un thread (QThread) pour ne pas bloquer l’UI. Voici un exemple minimal — formulaire de demande + liste prêts.

# file: loan_client.py
import sys
import requests
from PySide6.QtWidgets import (QApplication, QWidget, QVBoxLayout, QFormLayout,
                               QLineEdit, QPushButton, QLabel, QTableWidget, QTableWidgetItem)
from PySide6.QtCore import QThread, Signal

API_BASE = "https://ton-domaine/api"  # adapter
TOKEN = "Bearer ton_token_sanctum"

class ApiThread(QThread):
    finished_signal = Signal(object)

    def __init__(self, method, url, json=None):
        super().__init__()
        self.method = method
        self.url = url
        self.json = json

    def run(self):
        headers = {"Authorization": TOKEN, "Accept":"application/json"}
        try:
            if self.method == "GET":
                r = requests.get(self.url, headers=headers, timeout=20)
            else:
                r = requests.post(self.url, json=self.json, headers=headers, timeout=20)
            data = r.json()
        except Exception as e:
            data = {"error": str(e)}
        self.finished_signal.emit(data)

class LoanApp(QWidget):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Prêts - Gestion école")
        self.v = QVBoxLayout()
        form = QFormLayout()
        self.amount_in = QLineEdit()
        self.term_in = QLineEdit()
        self.interest_in = QLineEdit()
        form.addRow("Montant", self.amount_in)
        form.addRow("Terme (mois)", self.term_in)
        form.addRow("Taux annuel (%)", self.interest_in)
        btn_apply = QPushButton("Demander prêt")
        btn_apply.clicked.connect(self.request_loan)
        self.v.addLayout(form)
        self.v.addWidget(btn_apply)
        self.status_label = QLabel("")
        self.v.addWidget(self.status_label)

        self.table = QTableWidget(0,5)
        self.table.setHorizontalHeaderLabels(["ID","Montant","Term","Mensualité","Statut"])
        self.v.addWidget(self.table)
        btn_refresh = QPushButton("Rafraîchir prêts")
        btn_refresh.clicked.connect(self.load_loans)
        self.v.addWidget(btn_refresh)

        self.setLayout(self.v)
        self.load_loans()

    def request_loan(self):
        payload = {
            "amount": float(self.amount_in.text()),
            "term_months": int(self.term_in.text()),
            "interest_rate": float(self.interest_in.text() or 0)
        }
        self.status_label.setText("Envoi...")
        t = ApiThread("POST", f"{API_BASE}/loans", json=payload)
        t.finished_signal.connect(self.on_apply_done)
        t.start()

    def on_apply_done(self, data):
        if data and not data.get("error"):
            self.status_label.setText("Demande envoyée")
            self.load_loans()
        else:
            self.status_label.setText(f"Erreur: {data.get('error') if data else '??'}")

    def load_loans(self):
        t = ApiThread("GET", f"{API_BASE}/loans")
        t.finished_signal.connect(self.on_loans_loaded)
        t.start()

    def on_loans_loaded(self, data):
        if isinstance(data, dict) and data.get('error'):
            self.status_label.setText("Erreur: " + data.get('error'))
            return
        loans = data if isinstance(data, list) else data.get('data', data)
        self.table.setRowCount(0)
        for loan in loans:
            r = self.table.rowCount()
            self.table.insertRow(r)
            self.table.setItem(r,0,QTableWidgetItem(str(loan.get('id'))))
            self.table.setItem(r,1,QTableWidgetItem(str(loan.get('amount'))))
            self.table.setItem(r,2,QTableWidgetItem(str(loan.get('term_months'))))
            self.table.setItem(r,3,QTableWidgetItem(str(loan.get('monthly_payment') or '-')))
            self.table.setItem(r,4,QTableWidgetItem(loan.get('status')))

> En production : gère pagination, erreurs, refresh après paiement, confirmation utilisateur, et utilise QNetworkAccessManager ou aiohttp + asyncio pour meilleure réactivité.
